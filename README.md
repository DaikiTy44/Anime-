# Attack on Titan Quiz
print("Attack on Titan Quiz!")

questions = [
    "What is the name of the main character in Attack on Titan?",
    "Which titan is known for its ability to control other titans?",
    "What is the name of the city where the story begins?",
    "Who is the leader of the Survey Corps?",
    "What is Eren Yeager's ultimate goal?",
]

answers = [
    "Eren Yeager",
    "Colossal Titan",
    "Wall Maria",
    "Erwin Smith",
    "To defeat the Titans and reclaim the world",
]

score = 0
for i in range(len(questions)):
    print(questions[i])
    user_answer = input("Your answer: ")
    if user_answer.lower() == answers[i].lower():
        print("Correct!")
        score += 1
    else:
        print(f"Incorrect. The correct answer is: {answers[i]}")

print(f"\nYou got {score} out of {len(questions)} questions correct!")
